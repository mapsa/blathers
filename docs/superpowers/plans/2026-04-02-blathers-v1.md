# Blathers v1.0 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a pip-installable Python CLI that parses OWL/SHACL ontologies, validates them with five validators, weaves auto-extracted definitions with Markdown sidecars, and renders a static HTML documentation site with interactive diagrams and content negotiation configs.

**Architecture:** Python CLI (Click) uses rdflib/pyshacl to parse and validate ontologies, then extracts terms into a JSON manifest (`site-data.json`). A Jinja2 HTML renderer consumes the manifest to produce a self-contained static site. The manifest decouples extraction from presentation — renderers are swappable.

**Tech Stack:** Python 3.10+, Click (CLI), rdflib (RDF parsing), pyshacl (SHACL validation), PyYAML (config), python-frontmatter (sidecars), markdown (Markdown→HTML), Jinja2 (HTML templates), Pydantic (config validation)

**Spec:** `docs/superpowers/specs/2026-04-02-blathers-design.md` in the prism-ontology repo

```python
"""Configuration loading and validation for Blathers."""

from __future__ import annotations

from pathlib import Path
from typing import Optional

import yaml
from pydantic import BaseModel, Field


class ImportConfig(BaseModel):
    uri: str
    prefix: str
    path: Optional[str] = None


class MetadataConfig(BaseModel):
    title: str
    version: str
    license: str
    namespace: str
    prefix: str


class ValidationRules(BaseModel):
    shacl: bool = True
    consistency: bool = True
    completeness: bool = True
    conventions: bool = True
    overlap: bool = True


class ValidationConfig(BaseModel):
    fail_on: str = "error"
    rules: ValidationRules = Field(default_factory=ValidationRules)
    overlap: Optional[dict] = None


class ConnegConfig(BaseModel):
    generate: list[str] = Field(default_factory=list)
    base_uri: str = ""
    formats: list[str] = Field(default_factory=lambda: ["html", "ttl"])


class BlathersConfig(BaseModel):
    ontology: Path
    shacl: list[Path] = Field(default_factory=list)
    examples: list[str] = Field(default_factory=list)
    sidecars: Path = Path("sidecars/")
    figures: Path = Path("figures/")
    output: Path = Path("dist/")
    metadata: MetadataConfig
    imports: list[ImportConfig] = Field(default_factory=list)
    conneg: ConnegConfig = Field(default_factory=ConnegConfig)
    validation: ValidationConfig = Field(default_factory=ValidationConfig)

    _config_dir: Path = Path(".")

    class Config:
        arbitrary_types_allowed = True

    def resolve_path(self, rel: Path) -> Path:
        """Resolve a path relative to the config file's directory."""
        return self._config_dir / rel


def load_config(path: Path) -> BlathersConfig:
    """Load and validate a blathers.yaml config file."""
    path = Path(path)
    if not path.exists():
        raise FileNotFoundError(f"Config file not found: {path}")

    with open(path) as f:
        raw = yaml.safe_load(f)

    config = BlathersConfig(**raw)
    # Store the config directory for path resolution
    object.__setattr__(config, "_config_dir", path.parent)
    return config
```

```python
"""RDF/OWL ontology extraction using rdflib."""

from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path

from rdflib import Graph, Namespace, URIRef
from rdflib.namespace import DCTERMS, OWL, RDF, RDFS

SH = Namespace("http://www.w3.org/ns/shacl#")
VANN = Namespace("http://purl.org/vocab/vann/")


def _local_name(iri: str) -> str:
    """Extract the local name from an IRI (after # or last /)."""
    if "#" in iri:
        return iri.split("#")[-1]
    return iri.rsplit("/", 1)[-1]


def _str_or_none(val) -> str | None:
    return str(val) if val is not None else None


@dataclass
class ExtractedClass:
    iri: str
    local_name: str
    label: str | None = None
    comment: str | None = None
    superclasses: list[str] = field(default_factory=list)
    subclasses: list[str] = field(default_factory=list)
    properties: list[str] = field(default_factory=list)


@dataclass
class ExtractedProperty:
    iri: str
    local_name: str
    label: str | None = None
    comment: str | None = None
    domain: str | None = None
    range: str | None = None
    prop_type: str = "ObjectProperty"


@dataclass
class ShapeConstraint:
    path: str | None = None
    min_count: int | None = None
    max_count: int | None = None
    node_class: str | None = None


@dataclass
class ExtractedShape:
    iri: str
    local_name: str
    target_class: str | None = None
    constraints: list[ShapeConstraint] = field(default_factory=list)


@dataclass
class OntologyData:
    namespace: str
    metadata: dict[str, str]
    classes: list[ExtractedClass]
    properties: list[ExtractedProperty]
    shapes: list[ExtractedShape]
    graph: Graph


def _find_ontology_iri(g: Graph) -> URIRef | None:
    """Find the ontology IRI in the graph."""
    for s in g.subjects(RDF.type, OWL.Ontology):
        return s
    return None


def _extract_metadata(g: Graph, ont_iri: URIRef | None) -> dict[str, str]:
    """Extract ontology-level metadata."""
    meta: dict[str, str] = {}
    if ont_iri is None:
        return meta

    for pred, key in [
        (RDFS.label, "title"),
        (RDFS.comment, "description"),
        (OWL.versionInfo, "version"),
        (DCTERMS.license, "license"),
        (DCTERMS.creator, "creator"),
        (DCTERMS.title, "title"),
        (VANN.preferredNamespacePrefix, "prefix"),
        (VANN.preferredNamespaceUri, "namespace"),
    ]:
        for obj in g.objects(ont_iri, pred):
            val = str(obj)
            if key not in meta or key == "title":
                meta[key] = val
    return meta


def _extract_classes(g: Graph, namespace: str) -> list[ExtractedClass]:
    """Extract all OWL classes in the ontology namespace."""
    classes = []
    for cls_iri in g.subjects(RDF.type, OWL.Class):
        iri_str = str(cls_iri)
        if not iri_str.startswith(namespace):
            continue
        label = _str_or_none(g.value(cls_iri, RDFS.label))
        comment = _str_or_none(g.value(cls_iri, RDFS.comment))
        supers = [str(s) for s in g.objects(cls_iri, RDFS.subClassOf) if isinstance(s, URIRef)]
        classes.append(ExtractedClass(
            iri=iri_str,
            local_name=_local_name(iri_str),
            label=label,
            comment=comment,
            superclasses=supers,
        ))

    # Populate subclasses by inverse lookup
    iri_set = {c.iri for c in classes}
    for cls in classes:
        for other in classes:
            if cls.iri in other.superclasses:
                cls.subclasses.append(other.iri)

    return classes


def _extract_properties(g: Graph, namespace: str) -> list[ExtractedProperty]:
    """Extract all OWL properties in the ontology namespace."""
    props = []
    prop_types = [
        (OWL.ObjectProperty, "ObjectProperty"),
        (OWL.DatatypeProperty, "DatatypeProperty"),
        (OWL.AnnotationProperty, "AnnotationProperty"),
    ]
    seen = set()
    for rdf_type, type_name in prop_types:
        for prop_iri in g.subjects(RDF.type, rdf_type):
            iri_str = str(prop_iri)
            if not iri_str.startswith(namespace) or iri_str in seen:
                continue
            seen.add(iri_str)
            label = _str_or_none(g.value(prop_iri, RDFS.label))
            comment = _str_or_none(g.value(prop_iri, RDFS.comment))
            domain = _str_or_none(g.value(prop_iri, RDFS.domain))
            range_ = _str_or_none(g.value(prop_iri, RDFS.range))
            props.append(ExtractedProperty(
                iri=iri_str,
                local_name=_local_name(iri_str),
                label=label,
                comment=comment,
                domain=domain,
                range=range_,
                prop_type=type_name,
            ))
    return props


def _extract_shapes(g: Graph) -> list[ExtractedShape]:
    """Extract SHACL shapes."""
    shapes = []
    for shape_iri in g.subjects(RDF.type, SH.NodeShape):
        iri_str = str(shape_iri)
        target = _str_or_none(g.value(shape_iri, SH.targetClass))

        constraints = []
        for prop_shape in g.objects(shape_iri, SH.property):
            path = _str_or_none(g.value(prop_shape, SH.path))
            min_c = g.value(prop_shape, SH.minCount)
            max_c = g.value(prop_shape, SH.maxCount)
            cls = _str_or_none(g.value(prop_shape, SH["class"]))
            constraints.append(ShapeConstraint(
                path=path,
                min_count=int(min_c) if min_c is not None else None,
                max_count=int(max_c) if max_c is not None else None,
                node_class=cls,
            ))

        shapes.append(ExtractedShape(
            iri=iri_str,
            local_name=_local_name(iri_str),
            target_class=target,
            constraints=constraints,
        ))
    return shapes


def _populate_class_properties(classes: list[ExtractedClass], properties: list[ExtractedProperty]) -> None:
    """Link properties to their domain classes."""
    class_map = {c.iri: c for c in classes}
    for prop in properties:
        if prop.domain and prop.domain in class_map:
            class_map[prop.domain].properties.append(prop.iri)


def _detect_namespace(g: Graph, ont_iri: URIRef | None) -> str:
    """Detect the ontology namespace from VANN annotation or ontology IRI."""
    if ont_iri is not None:
        ns = g.value(ont_iri, VANN.preferredNamespaceUri)
        if ns:
            return str(ns)
        iri_str = str(ont_iri)
        if iri_str.endswith("#") or iri_str.endswith("/"):
            return iri_str
        return iri_str + "#"
    return ""


def extract_ontology(
    ontology_path: Path,
    shacl_paths: list[Path] | None = None,
) -> OntologyData:
    """Parse an OWL ontology and optional SHACL shapes, extracting all terms."""
    g = Graph()
    g.parse(str(ontology_path), format="turtle")

    if shacl_paths:
        for sp in shacl_paths:
            g.parse(str(sp), format="turtle")

    ont_iri = _find_ontology_iri(g)
    namespace = _detect_namespace(g, ont_iri)
    metadata = _extract_metadata(g, ont_iri)
    classes = _extract_classes(g, namespace)
    properties = _extract_properties(g, namespace)
    shapes = _extract_shapes(g)
    _populate_class_properties(classes, properties)

    return OntologyData(
        namespace=namespace,
        metadata=metadata,
        classes=classes,
        properties=properties,
        shapes=shapes,
        graph=g,
    )
```

```python
"""Sidecar Markdown file parsing."""

from __future__ import annotations

from dataclasses import dataclass
from pathlib import Path

import frontmatter
import markdown


@dataclass
class Sidecar:
    filename: str
    path: Path
    term: str | None
    section: str | None
    order: int
    body: str
    html: str
    is_narrative: bool


def _parse_sidecar(path: Path) -> Sidecar:
    """Parse a single Markdown sidecar with YAML frontmatter."""
    post = frontmatter.load(str(path))
    body = post.content
    html = markdown.markdown(body, extensions=["tables", "fenced_code"])
    filename = path.name
    is_narrative = filename.startswith("_")

    return Sidecar(
        filename=filename,
        path=path,
        term=post.get("term"),
        section=post.get("section"),
        order=post.get("order", 999),
        body=body,
        html=html,
        is_narrative=is_narrative,
    )


def load_sidecars(sidecars_dir: Path) -> list[Sidecar]:
    """Load all .md sidecar files from a directory."""
    sidecars_dir = Path(sidecars_dir)
    if not sidecars_dir.is_dir():
        return []

    sidecars = []
    for md_file in sorted(sidecars_dir.glob("*.md")):
        sidecars.append(_parse_sidecar(md_file))

    return sorted(sidecars, key=lambda s: s.order)
```

```python
"""SHACL validator — runs pyshacl against example instances."""

from __future__ import annotations

from pathlib import Path

from rdflib import Graph

from blathers.validators.base import Severity, ValidationResult


class ShaclValidator:
    """Validates example instances against SHACL shapes."""

    def __init__(
        self,
        ontology_path: Path,
        shacl_paths: list[Path],
        example_paths: list[Path],
    ) -> None:
        self.ontology_path = ontology_path
        self.shacl_paths = shacl_paths
        self.example_paths = example_paths

    def validate(self) -> list[ValidationResult]:
        if not self.example_paths:
            return []

        # Build the shapes graph (ontology + SHACL)
        shapes_graph = Graph()
        shapes_graph.parse(str(self.ontology_path), format="turtle")
        for sp in self.shacl_paths:
            shapes_graph.parse(str(sp), format="turtle")

        results: list[ValidationResult] = []

        for example_path in self.example_paths:
            data_graph = Graph()
            data_graph.parse(str(example_path), format="turtle")
            results.extend(self._validate_one(data_graph, shapes_graph))

        return results

    def _validate_one(
        self, data_graph: Graph, shapes_graph: Graph
    ) -> list[ValidationResult]:
        from pyshacl import validate as shacl_validate

        conforms, results_graph, results_text = shacl_validate(
            data_graph,
            shacl_graph=shapes_graph,
            inference="none",
            abort_on_first=False,
        )

        if conforms:
            return []

        results: list[ValidationResult] = []
        # Parse the results text for individual violations
        for line in results_text.strip().split("\n"):
            line = line.strip()
            if line and not line.startswith("Validation Report"):
                if "Violation" in line or "Result" in line or line.startswith("Message:"):
                    continue

        # Use the results graph for structured extraction
        from rdflib.namespace import SH

        SH_NS = SH
        for result_node in results_graph.subjects(
            predicate=None, object=SH_NS.ValidationResult
        ):
            message_val = results_graph.value(result_node, SH_NS.resultMessage)
            focus = results_graph.value(result_node, SH_NS.focusNode)
            severity_uri = results_graph.value(result_node, SH_NS.resultSeverity)

            severity = Severity.ERROR
            if severity_uri and "Warning" in str(severity_uri):
                severity = Severity.WARNING

            msg = str(message_val) if message_val else "SHACL violation"
            if focus:
                msg = f"{focus} — {msg}"

            results.append(ValidationResult(
                validator="shacl",
                severity=severity,
                message=msg,
                term=str(focus) if focus else None,
            ))

        # Fallback: if no structured results but validation failed, use text
        if not results and not conforms:
            results.append(ValidationResult(
                validator="shacl",
                severity=Severity.ERROR,
                message=results_text.strip()[:200],
            ))

        return results
```

```python
"""Consistency validator — dangling refs, orphan terms, missing labels."""

from __future__ import annotations

from rdflib import Graph, URIRef
from rdflib.namespace import OWL, RDF, RDFS

from blathers.validators.base import Severity, ValidationResult


class ConsistencyValidator:
    """Checks ontology internal consistency."""

    def __init__(self, graph: Graph, namespace: str) -> None:
        self.graph = graph
        self.namespace = namespace

    def validate(self) -> list[ValidationResult]:
        results: list[ValidationResult] = []
        results.extend(self._check_dangling_references())
        results.extend(self._check_missing_labels())
        results.extend(self._check_orphan_terms())
        return results

    def _local_classes(self) -> set[str]:
        return {
            str(s)
            for s in self.graph.subjects(RDF.type, OWL.Class)
            if str(s).startswith(self.namespace)
        }

    def _local_properties(self) -> set[str]:
        prop_types = [OWL.ObjectProperty, OWL.DatatypeProperty, OWL.AnnotationProperty]
        props = set()
        for pt in prop_types:
            for s in self.graph.subjects(RDF.type, pt):
                if str(s).startswith(self.namespace):
                    props.add(str(s))
        return props

    def _all_local_terms(self) -> set[str]:
        return self._local_classes() | self._local_properties()

    def _local_name(self, iri: str) -> str:
        if "#" in iri:
            return iri.split("#")[-1]
        return iri.rsplit("/", 1)[-1]

    def _check_dangling_references(self) -> list[ValidationResult]:
        """Check domain/range references to undefined classes."""
        results = []
        local_classes = self._local_classes()

        # Collect all known classes (local + external via imports)
        all_classes = set()
        for s in self.graph.subjects(RDF.type, OWL.Class):
            all_classes.add(str(s))

        for prop_iri in self._local_properties():
            prop_ref = URIRef(prop_iri)
            for pred_name, pred in [("domain", RDFS.domain), ("range", RDFS.range)]:
                obj = self.graph.value(prop_ref, pred)
                if obj is None:
                    continue
                obj_str = str(obj)
                # Only flag if it looks like a class IRI (not XSD, etc.) and isn't defined
                if obj_str.startswith(self.namespace) and obj_str not in all_classes:
                    results.append(ValidationResult(
                        validator="consistency",
                        severity=Severity.ERROR,
                        message=f"{self._local_name(prop_iri)} — {pred_name} {self._local_name(obj_str)} not defined",
                        term=prop_iri,
                    ))
        return results

    def _check_missing_labels(self) -> list[ValidationResult]:
        """Check that all terms have rdfs:label and rdfs:comment."""
        results = []
        for term_iri in self._all_local_terms():
            ref = URIRef(term_iri)
            name = self._local_name(term_iri)
            if not self.graph.value(ref, RDFS.label):
                results.append(ValidationResult(
                    validator="consistency",
                    severity=Severity.WARNING,
                    message=f"{name} — missing rdfs:label",
                    term=term_iri,
                ))
            if not self.graph.value(ref, RDFS.comment):
                results.append(ValidationResult(
                    validator="consistency",
                    severity=Severity.WARNING,
                    message=f"{name} — missing rdfs:comment",
                    term=term_iri,
                ))
        return results

    def _check_orphan_terms(self) -> list[ValidationResult]:
        """Check for terms that are defined but never referenced by other terms."""
        results = []
        local_classes = self._local_classes()
        local_props = self._local_properties()

        # A class is referenced if it appears as:
        # - superclass/subclass of another class
        # - domain/range of a property
        # - target of a SHACL shape
        referenced_classes: set[str] = set()
        for cls_iri in local_classes:
            for obj in self.graph.objects(URIRef(cls_iri), RDFS.subClassOf):
                if str(obj).startswith(self.namespace):
                    referenced_classes.add(str(obj))
        for prop_iri in local_props:
            domain = self.graph.value(URIRef(prop_iri), RDFS.domain)
            range_ = self.graph.value(URIRef(prop_iri), RDFS.range)
            if domain:
                referenced_classes.add(str(domain))
            if range_:
                referenced_classes.add(str(range_))

        for cls_iri in local_classes:
            # Also check if it has subclasses
            has_subclass = any(
                str(s).startswith(self.namespace)
                for s in self.graph.subjects(RDFS.subClassOf, URIRef(cls_iri))
            )
            if cls_iri not in referenced_classes and not has_subclass:
                results.append(ValidationResult(
                    validator="consistency",
                    severity=Severity.WARNING,
                    message=f"{self._local_name(cls_iri)} — orphan class (defined but never referenced)",
                    term=cls_iri,
                ))
        return results
```

```python
"""Completeness validator — checks sidecar coverage of ontology terms."""

from __future__ import annotations

from blathers.sidecars import Sidecar
from blathers.validators.base import Severity, ValidationResult


class CompletenessValidator:
    """Checks that all terms have sidecars and all sidecars reference real terms."""

    def __init__(
        self,
        class_iris: set[str],
        property_iris: set[str],
        sidecars: list[Sidecar],
        namespace: str,
        prefix: str,
    ) -> None:
        self.class_iris = class_iris
        self.property_iris = property_iris
        self.sidecars = sidecars
        self.namespace = namespace
        self.prefix = prefix

    def _expand_prefixed(self, term: str) -> str:
        """Expand prefix:LocalName to full IRI."""
        if ":" in term and not term.startswith("http"):
            prefix, local = term.split(":", 1)
            if prefix == self.prefix:
                return self.namespace + local
        return term

    def _local_name(self, iri: str) -> str:
        if "#" in iri:
            return iri.split("#")[-1]
        return iri.rsplit("/", 1)[-1]

    def validate(self) -> list[ValidationResult]:
        results: list[ValidationResult] = []
        all_iris = self.class_iris | self.property_iris

        # Check: terms without sidecars
        documented_iris = set()
        for sc in self.sidecars:
            if sc.term:
                documented_iris.add(self._expand_prefixed(sc.term))

        for iri in all_iris:
            if iri not in documented_iris:
                results.append(ValidationResult(
                    validator="completeness",
                    severity=Severity.WARNING,
                    message=f"{self._local_name(iri)} — no sidecar documentation",
                    term=iri,
                ))

        # Check: sidecars referencing non-existent terms
        for sc in self.sidecars:
            if sc.term:
                expanded = self._expand_prefixed(sc.term)
                if expanded not in all_iris:
                    results.append(ValidationResult(
                        validator="completeness",
                        severity=Severity.WARNING,
                        message=f"{sc.term} — sidecar references term not found in ontology",
                        term=expanded,
                    ))

        return results
```

```python
"""Convention validator — naming, metadata, prefix hygiene."""

from __future__ import annotations

import re

from rdflib import Graph, URIRef
from rdflib.namespace import DCTERMS, OWL, RDF, RDFS

from blathers.validators.base import Severity, ValidationResult


def _local_name(iri: str) -> str:
    if "#" in iri:
        return iri.split("#")[-1]
    return iri.rsplit("/", 1)[-1]


def _is_pascal_case(name: str) -> bool:
    return bool(re.match(r"^[A-Z][a-zA-Z0-9]*$", name))


def _is_camel_case(name: str) -> bool:
    return bool(re.match(r"^[a-z][a-zA-Z0-9]*$", name))


class ConventionValidator:
    """Checks naming conventions and required metadata."""

    def __init__(self, graph: Graph, namespace: str) -> None:
        self.graph = graph
        self.namespace = namespace

    def validate(self) -> list[ValidationResult]:
        results: list[ValidationResult] = []
        results.extend(self._check_naming())
        results.extend(self._check_metadata())
        return results

    def _check_naming(self) -> list[ValidationResult]:
        """Classes should be PascalCase, properties camelCase."""
        results = []

        for cls_iri in self.graph.subjects(RDF.type, OWL.Class):
            iri_str = str(cls_iri)
            if not iri_str.startswith(self.namespace):
                continue
            name = _local_name(iri_str)
            if not _is_pascal_case(name):
                results.append(ValidationResult(
                    validator="conventions",
                    severity=Severity.WARNING,
                    message=f"{name} — class names should be PascalCase",
                    term=iri_str,
                ))

        prop_types = [OWL.ObjectProperty, OWL.DatatypeProperty]
        seen = set()
        for pt in prop_types:
            for prop_iri in self.graph.subjects(RDF.type, pt):
                iri_str = str(prop_iri)
                if not iri_str.startswith(self.namespace) or iri_str in seen:
                    continue
                seen.add(iri_str)
                name = _local_name(iri_str)
                if not _is_camel_case(name):
                    results.append(ValidationResult(
                        validator="conventions",
                        severity=Severity.WARNING,
                        message=f"{name} — property names should be camelCase",
                        term=iri_str,
                    ))

        return results

    def _check_metadata(self) -> list[ValidationResult]:
        """Ontology must declare version, license, creator, title."""
        results = []

        ont_iri = None
        for s in self.graph.subjects(RDF.type, OWL.Ontology):
            ont_iri = s
            break

        if ont_iri is None:
            results.append(ValidationResult(
                validator="conventions",
                severity=Severity.ERROR,
                message="No owl:Ontology declaration found",
            ))
            return results

        checks = [
            (OWL.versionInfo, "version"),
            (DCTERMS.license, "license"),
            (DCTERMS.creator, "creator"),
        ]
        # Title can be dcterms:title or rdfs:label
        has_title = (
            self.graph.value(ont_iri, DCTERMS.title) is not None
            or self.graph.value(ont_iri, RDFS.label) is not None
        )
        if not has_title:
            results.append(ValidationResult(
                validator="conventions",
                severity=Severity.WARNING,
                message="Ontology missing title (dcterms:title or rdfs:label)",
            ))

        for predicate, name in checks:
            if self.graph.value(ont_iri, predicate) is None:
                results.append(ValidationResult(
                    validator="conventions",
                    severity=Severity.WARNING,
                    message=f"Ontology missing {name} ({predicate})",
                ))

        return results
```

```python
"""Import resolution and caching for ontology dependencies."""

from __future__ import annotations

import re
from pathlib import Path

from rdflib import Graph

from blathers.config import ImportConfig


def uri_to_cache_filename(uri: str) -> str:
    """Convert a URI to a safe cache filename."""
    safe = re.sub(r"[^a-zA-Z0-9._-]", "_", uri.rstrip("#/"))
    return safe + ".ttl"


class ImportResolver:
    """Resolves imported ontologies from local paths, cache, or network."""

    def __init__(
        self,
        imports: list[ImportConfig],
        project_dir: Path,
        cache_dir: Path,
    ) -> None:
        self.imports = imports
        self.project_dir = project_dir
        self.cache_dir = cache_dir

    def resolve(self, imp: ImportConfig) -> Graph | None:
        """Resolve a single import. Returns a parsed Graph or None."""
        # 1. Local path (relative to project dir)
        if imp.path:
            local = self.project_dir / imp.path
            if local.exists():
                g = Graph()
                g.parse(str(local), format="turtle")
                return g

        # 2. Cache
        cache_file = self.cache_dir / uri_to_cache_filename(imp.uri)
        if cache_file.exists():
            g = Graph()
            g.parse(str(cache_file), format="turtle")
            return g

        # 3. Network fetch (try to dereference the IRI)
        try:
            g = Graph()
            g.parse(imp.uri)
            # Cache for future use
            self._cache_graph(g, imp.uri)
            return g
        except Exception:
            return None

    def resolve_all(self) -> dict[str, Graph]:
        """Resolve all imports. Returns {uri: Graph}."""
        resolved: dict[str, Graph] = {}
        for imp in self.imports:
            graph = self.resolve(imp)
            if graph is not None:
                resolved[imp.uri] = graph
        return resolved

    def fetch_and_cache(self, imp: ImportConfig) -> bool:
        """Fetch an import from network and cache it. Returns True on success."""
        try:
            g = Graph()
            if imp.path:
                local = self.project_dir / imp.path
                if local.exists():
                    g.parse(str(local), format="turtle")
                    self._cache_graph(g, imp.uri)
                    return True

            g.parse(imp.uri)
            self._cache_graph(g, imp.uri)
            return True
        except Exception:
            return False

    def _cache_graph(self, g: Graph, uri: str) -> None:
        """Write a graph to the cache directory."""
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        cache_file = self.cache_dir / uri_to_cache_filename(uri)
        g.serialize(str(cache_file), format="turtle")
```

```python
"""Overlap/redundancy validator — detects shadowed imported terms."""

from __future__ import annotations

from rdflib import Graph
from rdflib.namespace import OWL, RDF

from blathers.validators.base import Severity, ValidationResult


def _local_name(iri: str) -> str:
    if "#" in iri:
        return iri.split("#")[-1]
    return iri.rsplit("/", 1)[-1]


def _extract_term_local_names(graph: Graph, namespace: str | None = None) -> dict[str, str]:
    """Extract {local_name: full_iri} for all OWL classes and properties."""
    terms: dict[str, str] = {}
    type_preds = [OWL.Class, OWL.ObjectProperty, OWL.DatatypeProperty, OWL.AnnotationProperty]
    for type_pred in type_preds:
        for s in graph.subjects(RDF.type, type_pred):
            iri = str(s)
            if namespace and not iri.startswith(namespace):
                continue
            name = _local_name(iri)
            terms[name] = iri
    return terms


class OverlapValidator:
    """Detects terms that shadow or redefine imported ontology terms."""

    def __init__(
        self,
        local_graph: Graph,
        local_namespace: str,
        imported_graphs: dict[str, Graph],
        allowlist: set[str],
    ) -> None:
        self.local_graph = local_graph
        self.local_namespace = local_namespace
        self.imported_graphs = imported_graphs
        self.allowlist = allowlist

    def _expand_allowlist(self) -> set[str]:
        """Expand prefixed names in allowlist to local names."""
        names = set()
        for entry in self.allowlist:
            if ":" in entry:
                names.add(entry.split(":", 1)[1])
            else:
                names.add(entry)
        return names

    def validate(self) -> list[ValidationResult]:
        results: list[ValidationResult] = []
        allowed_names = self._expand_allowlist()

        local_terms = _extract_term_local_names(self.local_graph, self.local_namespace)

        for import_uri, imported_graph in self.imported_graphs.items():
            imported_terms = _extract_term_local_names(imported_graph)
            import_prefix = _local_name(import_uri.rstrip("#/"))

            for local_name, local_iri in local_terms.items():
                if local_name in allowed_names:
                    continue
                if local_name in imported_terms:
                    results.append(ValidationResult(
                        validator="overlap",
                        severity=Severity.WARNING,
                        message=(
                            f"{local_name} — shadows {imported_terms[local_name]} "
                            f"(imported via {import_uri})"
                        ),
                        term=local_iri,
                    ))

        return results
```

```python
"""JSON manifest generation — the contract between extraction and rendering."""

from __future__ import annotations

from blathers.config import BlathersConfig
from blathers.extract import OntologyData
from blathers.sidecars import Sidecar
from blathers.validators.base import Severity, ValidationResult


def _match_sidecar(term_iri: str, prefix: str, namespace: str, sidecars: list[Sidecar]) -> Sidecar | None:
    """Find a sidecar that matches a term IRI."""
    local_name = term_iri.split("#")[-1] if "#" in term_iri else term_iri.rsplit("/", 1)[-1]
    prefixed = f"{prefix}:{local_name}"
    for sc in sidecars:
        if sc.term == prefixed:
            return sc
        if sc.term == term_iri:
            return sc
        # Match by filename (MyClass.md → MyClass)
        if not sc.is_narrative and sc.filename.replace(".md", "") == local_name:
            return sc
    return None


def build_manifest(
    config: BlathersConfig,
    data: OntologyData,
    sidecars: list[Sidecar],
    validation_results: list[ValidationResult],
) -> dict:
    """Build the site-data.json manifest."""
    prefix = config.metadata.prefix
    namespace = config.metadata.namespace

    classes = []
    for cls in data.classes:
        sc = _match_sidecar(cls.iri, prefix, namespace, sidecars)
        classes.append({
            "iri": cls.iri,
            "local_name": cls.local_name,
            "label": cls.label,
            "comment": cls.comment,
            "superclasses": cls.superclasses,
            "subclasses": cls.subclasses,
            "properties": cls.properties,
            "sidecar": sc.html if sc else None,
        })

    properties = []
    for prop in data.properties:
        sc = _match_sidecar(prop.iri, prefix, namespace, sidecars)
        properties.append({
            "iri": prop.iri,
            "local_name": prop.local_name,
            "label": prop.label,
            "comment": prop.comment,
            "domain": prop.domain,
            "range": prop.range,
            "prop_type": prop.prop_type,
            "sidecar": sc.html if sc else None,
        })

    shapes = []
    for shape in data.shapes:
        shapes.append({
            "iri": shape.iri,
            "local_name": shape.local_name,
            "target_class": shape.target_class,
            "constraints": [
                {
                    "path": c.path,
                    "min_count": c.min_count,
                    "max_count": c.max_count,
                    "node_class": c.node_class,
                }
                for c in shape.constraints
            ],
        })

    sections = []
    for sc in sidecars:
        if sc.is_narrative:
            sections.append({
                "filename": sc.filename,
                "section": sc.section,
                "order": sc.order,
                "html": sc.html,
            })

    error_count = sum(1 for r in validation_results if r.severity == Severity.ERROR)
    warning_count = sum(1 for r in validation_results if r.severity == Severity.WARNING)

    return {
        "metadata": {
            "title": config.metadata.title,
            "version": config.metadata.version,
            "license": config.metadata.license,
            "namespace": config.metadata.namespace,
            "prefix": config.metadata.prefix,
        },
        "classes": classes,
        "properties": properties,
        "shapes": shapes,
        "sections": sections,
        "conneg": {
            "formats": config.conneg.formats,
            "base_uri": config.conneg.base_uri,
        },
        "validation_summary": {
            "errors": error_count,
            "warnings": warning_count,
        },
    }
```

```python
"""Ontology serialization to multiple RDF formats."""

from __future__ import annotations

import shutil
from pathlib import Path

from rdflib import Graph

FORMAT_MAP = {
    "ttl": ("turtle", "ontology.ttl"),
    "jsonld": ("json-ld", "ontology.jsonld"),
    "nt": ("nt", "ontology.nt"),
    "owl": ("xml", "ontology.owl"),
}


def serialize_ontology(source_path: Path, output_dir: Path, formats: list[str]) -> list[Path]:
    """Serialize the ontology into the requested formats."""
    output_dir.mkdir(parents=True, exist_ok=True)
    created: list[Path] = []

    g = Graph()
    g.parse(str(source_path), format="turtle")

    for fmt in formats:
        if fmt not in FORMAT_MAP:
            continue

        rdflib_format, filename = FORMAT_MAP[fmt]
        out_path = output_dir / filename

        if fmt == "ttl":
            # Copy source as-is to preserve formatting
            shutil.copy2(source_path, out_path)
        else:
            g.serialize(str(out_path), format=rdflib_format)

        created.append(out_path)

    return created
```

```python
"""Content negotiation config generation (Apache, nginx, w3id.org)."""

from __future__ import annotations

from pathlib import Path

MIME_TYPES = {
    "ttl": "text/turtle",
    "jsonld": "application/ld+json",
    "nt": "application/n-triples",
    "owl": "application/rdf+xml",
}

FILE_NAMES = {
    "ttl": "ontology.ttl",
    "jsonld": "ontology.jsonld",
    "nt": "ontology.nt",
    "owl": "ontology.owl",
}


def _generate_htaccess(output_dir: Path, formats: list[str]) -> Path:
    lines = ["RewriteEngine On", ""]
    for fmt in formats:
        if fmt in MIME_TYPES:
            mime = MIME_TYPES[fmt]
            filename = FILE_NAMES[fmt]
            # Escape + in regex
            mime_pattern = mime.replace("+", r"\+")
            lines.append(f"RewriteCond %{{HTTP_ACCEPT}} {mime_pattern}")
            lines.append(f"RewriteRule ^$ {filename} [R=303,L]")
            lines.append("")

    # Default to HTML
    lines.append("RewriteRule ^$ index.html [R=303,L]")
    lines.append("")

    path = output_dir / ".htaccess"
    path.write_text("\n".join(lines))
    return path


def _generate_nginx(output_dir: Path, base_uri: str, formats: list[str]) -> Path:
    # Extract the path from the base URI
    from urllib.parse import urlparse
    parsed = urlparse(base_uri)
    location = parsed.path or "/"

    lines = [f"location {location} {{"]
    for fmt in formats:
        if fmt in MIME_TYPES:
            mime = MIME_TYPES[fmt]
            filename = FILE_NAMES[fmt]
            mime_pattern = mime.replace("+", r"\+")
            lines.append(f'    if ($http_accept ~* "{mime_pattern}") {{ return 303 {location}/{filename}; }}')

    lines.append(f"    return 303 {location}/index.html;")
    lines.append("}")
    lines.append("")

    path = output_dir / "conneg.nginx.conf"
    path.write_text("\n".join(lines))
    return path


def _generate_w3id(output_dir: Path, base_uri: str, formats: list[str]) -> Path:
    from urllib.parse import urlparse
    parsed = urlparse(base_uri)
    # w3id.org redirects to the actual hosting location
    host_path = parsed.path.lstrip("/")

    lines = [
        "# w3id.org content negotiation config",
        f"# Generated by Blathers for {base_uri}",
        "",
        "RewriteEngine On",
        "",
    ]

    # Per-format redirects
    for fmt in formats:
        if fmt in MIME_TYPES:
            mime = MIME_TYPES[fmt]
            filename = FILE_NAMES[fmt]
            mime_pattern = mime.replace("+", r"\+")
            lines.append(f"RewriteCond %{{HTTP_ACCEPT}} {mime_pattern}")
            lines.append(f"RewriteRule ^$ {base_uri}/{filename} [R=303,L]")
            lines.append("")

    # Hash IRIs — redirect to HTML with fragment
    lines.append("# Hash IRIs: /prism#Term → /prism/index.html#Term")
    lines.append(f"RewriteRule ^$ {base_uri}/index.html [R=303,L]")
    lines.append("")

    path = output_dir / "w3id.htaccess"
    path.write_text("\n".join(lines))
    return path


def generate_conneg(
    output_dir: Path,
    generate: list[str],
    base_uri: str,
    formats: list[str],
) -> list[Path]:
    """Generate content negotiation configs."""
    output_dir.mkdir(parents=True, exist_ok=True)
    created: list[Path] = []

    if "htaccess" in generate:
        created.append(_generate_htaccess(output_dir, formats))
    if "nginx" in generate:
        created.append(_generate_nginx(output_dir, base_uri, formats))
    if "w3id" in generate:
        created.append(_generate_w3id(output_dir, base_uri, formats))

    return created
```

```python
"""SVG diagram link injection and diagram collection."""

from __future__ import annotations

import re
from pathlib import Path

DIAGRAM_EXTENSIONS = {".svg", ".png", ".jpg", ".jpeg"}


def inject_links(svg_content: str, term_names: set[str], base_path: str = "classes") -> str:
    """Inject clickable links into SVG elements whose IDs match ontology terms.

    Wraps matching elements in <a xlink:href="..."> tags.
    """
    if not term_names:
        return svg_content

    # For each term, find elements with matching id and wrap in <a>
    for term in term_names:
        # Match elements with id="TermName" — wrap the whole element in an <a> tag
        pattern = re.compile(
            rf'(<(?:rect|text|g|ellipse|circle|path|polygon|use)\s[^>]*id="{re.escape(term)}"[^>]*/?>)',
            re.DOTALL,
        )
        # Also match elements with closing tags
        pattern_with_close = re.compile(
            rf'(<(?:rect|text|g|ellipse|circle|path|polygon|use)\s[^>]*id="{re.escape(term)}"[^>]*>.*?</(?:rect|text|g|ellipse|circle|path|polygon|use)>)',
            re.DOTALL,
        )

        href = f"{base_path}/{term}.html"
        replacement = (
            f'<a href="{href}" style="cursor:pointer">'
            rf"\1"
            f"</a>"
        )

        # Try with-close first, then self-closing
        svg_content, n = pattern_with_close.subn(replacement, svg_content)
        if n == 0:
            svg_content = pattern.sub(replacement, svg_content)

    return svg_content


def collect_diagrams(figures_dir: Path) -> list[Path]:
    """Collect diagram files from the figures directory."""
    figures_dir = Path(figures_dir)
    if not figures_dir.is_dir():
        return []

    return sorted(
        p for p in figures_dir.iterdir()
        if p.suffix.lower() in DIAGRAM_EXTENSIONS
    )
```

```javascript
/* Blathers — search and theme toggle */
(function() {
    "use strict";

    /* Theme toggle */
    var toggle = document.getElementById("theme-toggle");
    if (toggle) {
        var saved = localStorage.getItem("blathers-theme");
        if (saved) document.documentElement.setAttribute("data-theme", saved);

        toggle.addEventListener("click", function() {
            var current = document.documentElement.getAttribute("data-theme") || "light";
            var next = current === "light" ? "dark" : "light";
            document.documentElement.setAttribute("data-theme", next);
            localStorage.setItem("blathers-theme", next);
            toggle.textContent = next === "dark" ? "☀️" : "🌙";
        });
    }

    /* Search */
    var searchInput = document.getElementById("search");
    var searchResults = document.getElementById("search-results");
    var searchIndex = null;

    function loadSearchIndex() {
        /* Search index is inlined as a global variable */
        if (typeof window.BLATHERS_SEARCH_INDEX !== "undefined") {
            searchIndex = window.BLATHERS_SEARCH_INDEX;
        }
    }

    function doSearch(query) {
        if (!searchIndex || !query) {
            searchResults.style.display = "none";
            return;
        }
        query = query.toLowerCase();
        var matches = searchIndex.filter(function(item) {
            return item.label.toLowerCase().indexOf(query) !== -1 ||
                   item.local_name.toLowerCase().indexOf(query) !== -1;
        }).slice(0, 15);

        if (matches.length === 0) {
            searchResults.style.display = "none";
            return;
        }

        searchResults.innerHTML = matches.map(function(m) {
            return '<a href="' + m.href + '">' + m.label + ' <small>(' + m.type + ')</small></a>';
        }).join("");
        searchResults.style.display = "block";
    }

    if (searchInput) {
        loadSearchIndex();
        searchInput.addEventListener("input", function() { doSearch(this.value); });
        document.addEventListener("click", function(e) {
            if (!searchInput.contains(e.target) && !searchResults.contains(e.target)) {
                searchResults.style.display = "none";
            }
        });
    }
})();
```

```python
"""Static HTML site renderer using Jinja2 templates."""

from __future__ import annotations

import json
import shutil
from pathlib import Path

from jinja2 import Environment, FileSystemLoader, select_autoescape

TEMPLATES_DIR = Path(__file__).parent / "templates"


def _local_name_filter(iri: str) -> str:
    """Jinja2 filter: extract local name from IRI."""
    if "#" in iri:
        return iri.split("#")[-1]
    return iri.rsplit("/", 1)[-1]


def _build_search_index(manifest: dict, root: str = "") -> list[dict]:
    """Build a flat search index for client-side search."""
    index = []
    for cls in manifest.get("classes", []):
        index.append({
            "label": cls.get("label") or cls["local_name"],
            "local_name": cls["local_name"],
            "type": "Class",
            "href": f"{root}classes/{cls['local_name']}.html",
        })
    for prop in manifest.get("properties", []):
        index.append({
            "label": prop.get("label") or prop["local_name"],
            "local_name": prop["local_name"],
            "type": "Property",
            "href": f"{root}properties/{prop['local_name']}.html",
        })
    return index


def render_site(manifest: dict, output_dir: Path) -> None:
    """Render the full static documentation site from a manifest."""
    output_dir.mkdir(parents=True, exist_ok=True)

    env = Environment(
        loader=FileSystemLoader(str(TEMPLATES_DIR)),
        autoescape=select_autoescape(["html"]),
    )
    env.filters["localname"] = _local_name_filter

    # Load CSS and JS
    css = (TEMPLATES_DIR / "style.css").read_text()
    js = (TEMPLATES_DIR / "blathers.js").read_text()

    # Build search index and inline it
    search_index = _build_search_index(manifest)
    search_js = f"window.BLATHERS_SEARCH_INDEX = {json.dumps(search_index)};\n{js}"

    metadata = manifest["metadata"]
    classes = manifest.get("classes", [])
    properties = manifest.get("properties", [])
    shapes = manifest.get("shapes", [])
    sections = manifest.get("sections", [])
    validation_summary = manifest.get("validation_summary", {"errors": 0, "warnings": 0})

    common = {
        "metadata": metadata,
        "classes": classes,
        "properties": properties,
        "shapes": shapes,
        "sections": sections,
        "css": css,
        "js": search_js,
        "root": "",
    }

    # Render index.html
    index_tpl = env.get_template("index.html")
    index_html = index_tpl.render(**common, validation_summary=validation_summary)
    (output_dir / "index.html").write_text(index_html)

    # Render class pages
    term_tpl = env.get_template("term.html")
    classes_dir = output_dir / "classes"
    classes_dir.mkdir(exist_ok=True)
    for cls in classes:
        html = term_tpl.render(**common, term=cls, term_type="Class", page_title=cls.get("label") or cls["local_name"])
        (classes_dir / f"{cls['local_name']}.html").write_text(html)

    # Render property pages
    props_dir = output_dir / "properties"
    props_dir.mkdir(exist_ok=True)
    for prop in properties:
        html = term_tpl.render(**common, term=prop, term_type="Property", page_title=prop.get("label") or prop["local_name"])
        (props_dir / f"{prop['local_name']}.html").write_text(html)

    # Render section pages
    section_tpl = env.get_template("section.html")
    sections_dir = output_dir / "sections"
    sections_dir.mkdir(exist_ok=True)
    for section in sections:
        section_name = section.get("section", "unnamed")
        html = section_tpl.render(**common, section_html=section["html"], page_title=section_name.title())
        (sections_dir / f"{section_name}.html").write_text(html)

    # Write assets
    assets_dir = output_dir / "assets"
    assets_dir.mkdir(exist_ok=True)
    (assets_dir / "style.css").write_text(css)
    (assets_dir / "blathers.js").write_text(search_js)

    # Write site-data.json
    (output_dir / "site-data.json").write_text(json.dumps(manifest, indent=2))

    # Create versioned snapshot
    version = metadata.get("version", "")
    if version:
        version_dir = output_dir / version
        if version_dir.exists():
            shutil.rmtree(version_dir)
        version_dir.mkdir()
        # Copy index to versioned dir
        shutil.copy2(output_dir / "index.html", version_dir / "index.html")
```

```python
from __future__ import annotations

import json
import sys
from glob import glob as globfn
from pathlib import Path

import click

from blathers import __version__


@click.group()
@click.version_option(version=__version__, prog_name="blathers")
def main() -> None:
    """Blathers — Ontology documentation & validation."""


@main.command()
@click.option("--dir", "project_dir", default=".", help="Directory to scaffold in.")
def init(project_dir: str) -> None:
    """Scaffold a new Blathers project."""
    project = Path(project_dir)
    project.mkdir(parents=True, exist_ok=True)

    # Create directories
    for d in ["sidecars", "figures"]:
        (project / d).mkdir(exist_ok=True)

    # Create default blathers.yaml
    config_path = project / "blathers.yaml"
    if not config_path.exists():
        config_path.write_text(
            "ontology: ontology/my-ontology.ttl\n"
            "shacl: []\n"
            "examples: []\n"
            "sidecars: sidecars/\n"
            "figures: figures/\n"
            "output: dist/\n"
            "\n"
            "metadata:\n"
            '  title: "My Ontology"\n'
            '  version: "0.1.0"\n'
            '  license: "MIT"\n'
            '  namespace: "http://example.org/my-ontology#"\n'
            '  prefix: ex\n'
            "\n"
            "imports: []\n"
            "\n"
            "conneg:\n"
            "  generate: [htaccess]\n"
            '  base_uri: "http://example.org/my-ontology"\n'
            "  formats: [html, ttl, jsonld]\n"
            "\n"
            "validation:\n"
            "  fail_on: error\n"
            "  rules:\n"
            "    shacl: true\n"
            "    consistency: true\n"
            "    completeness: true\n"
            "    conventions: true\n"
            "    overlap: true\n"
        )

    # Create index sidecar
    index_sidecar = project / "sidecars" / "_index.md"
    if not index_sidecar.exists():
        index_sidecar.write_text(
            "---\n"
            "section: introduction\n"
            "order: 1\n"
            "---\n"
            "\n"
            "# My Ontology\n"
            "\n"
            "Welcome to the documentation.\n"
        )

    click.echo(f"Scaffolded Blathers project in {project}")
    click.echo("  blathers.yaml — edit this to configure your project")
    click.echo("  sidecars/ — add Markdown documentation here")
    click.echo("  figures/ — add diagrams here")


@main.command()
@click.option("--config", "config_path", default="blathers.yaml", help="Path to blathers.yaml")
def validate(config_path: str) -> None:
    """Run all validators."""
    from blathers.config import load_config
    from blathers.extract import extract_ontology
    from blathers.imports import ImportResolver
    from blathers.sidecars import load_sidecars
    from blathers.validators.base import Severity, ValidationResult
    from blathers.validators.shacl import ShaclValidator
    from blathers.validators.consistency import ConsistencyValidator
    from blathers.validators.completeness import CompletenessValidator
    from blathers.validators.conventions import ConventionValidator
    from blathers.validators.overlap import OverlapValidator

    config = load_config(Path(config_path))
    all_results: list[ValidationResult] = []

    # Extract ontology
    ontology_path = config.resolve_path(config.ontology)
    shacl_paths = [config.resolve_path(p) for p in config.shacl]
    data = extract_ontology(ontology_path, shacl_paths=shacl_paths)

    # Resolve example paths (support globs)
    example_paths: list[Path] = []
    for pattern in config.examples:
        for match in globfn(str(config.resolve_path(Path(pattern)))):
            example_paths.append(Path(match))

    rules = config.validation.rules

    # SHACL
    if rules.shacl and shacl_paths and example_paths:
        click.echo("  SHACL", nl=False)
        v = ShaclValidator(ontology_path, shacl_paths, example_paths)
        results = v.validate()
        all_results.extend(results)
        if not results:
            click.echo(f"       {len(example_paths)} instances validated against {len(shacl_paths)} shape files")
        else:
            click.echo("")
            for r in results:
                click.echo(f"    {r.severity.value.upper():8s} {r.message}")

    # Consistency
    if rules.consistency:
        click.echo("  Consistency", nl=False)
        v = ConsistencyValidator(data.graph, namespace=data.namespace)
        results = v.validate()
        all_results.extend(results)
        if not results:
            click.echo(" OK")
        else:
            click.echo("")
            for r in results:
                click.echo(f"    {r.severity.value.upper():8s} {r.message}")

    # Completeness
    if rules.completeness:
        click.echo("  Completeness", nl=False)
        sidecars = load_sidecars(config.resolve_path(config.sidecars))

    if rules.conventions:

                click.echo(f"    {r.severity.value.upper():8s} {r.message}")

        click.echo("  Overlap", nl=False)

            imports=config.imports,
            project_dir=config._config_dir,

            click.echo("    OK")

@main.command()
@click.option("--config", "config_path", default="blathers.yaml", help="Path to blathers.yaml")
def build(config_path: str) -> None:
    """Validate, extract, and render static site."""
    from blathers.config import load_config
    from blathers.conneg import generate_conneg
    from blathers.diagrams import collect_diagrams, inject_links
    from blathers.extract import extract_ontology
    from blathers.imports import ImportResolver
    from blathers.manifest import build_manifest
    from blathers.render import render_site
    from blathers.serialize import serialize_ontology
    from blathers.sidecars import load_sidecars
    from blathers.validators.base import Severity, ValidationResult
    from blathers.validators.shacl import ShaclValidator
    from blathers.validators.consistency import ConsistencyValidator
    from blathers.validators.completeness import CompletenessValidator
    from blathers.validators.conventions import ConventionValidator
    from blathers.validators.overlap import OverlapValidator

    config = load_config(Path(config_path))
    output_dir = config.resolve_path(config.output)

    click.echo(f"Building {config.metadata.title} v{config.metadata.version}")

    # 1. Extract
    click.echo("  Extracting ontology...")
    ontology_path = config.resolve_path(config.ontology)
    shacl_paths = [config.resolve_path(p) for p in config.shacl]
    data = extract_ontology(ontology_path, shacl_paths=shacl_paths)

    # 2. Load sidecars
    click.echo("  Loading sidecars...")
    sidecars = load_sidecars(config.resolve_path(config.sidecars))

    # 3. Validate (non-fatal during build — collect results)
    click.echo("  Validating...")
    all_results: list[ValidationResult] = []

    example_paths: list[Path] = []
    for pattern in config.examples:
        for match in globfn(str(config.resolve_path(Path(pattern)))):
            example_paths.append(Path(match))

    rules = config.validation.rules
    if rules.shacl and shacl_paths and example_paths:
        all_results.extend(ShaclValidator(ontology_path, shacl_paths, example_paths).validate())
    if rules.consistency:
        all_results.extend(ConsistencyValidator(data.graph, data.namespace).validate())
    if rules.completeness:
        all_results.extend(CompletenessValidator(
            class_iris={c.iri for c in data.classes},
            property_iris={p.iri for p in data.properties},
            sidecars=sidecars,
            namespace=data.namespace,
            prefix=config.metadata.prefix,
        ).validate())
    if rules.conventions:
        all_results.extend(ConventionValidator(data.graph, data.namespace).validate())
    if rules.overlap and config.imports:
        resolver = ImportResolver(config.imports, config._config_dir, config._config_dir / ".blathers" / "imports")
        imported_graphs = resolver.resolve_all()
        allowlist = set()
        if config.validation.overlap and isinstance(config.validation.overlap, dict):
            allowlist = set(config.validation.overlap.get("allow", []))
        all_results.extend(OverlapValidator(data.graph, data.namespace, imported_graphs, allowlist).validate())

    errors = sum(1 for r in all_results if r.severity == Severity.ERROR)
    warnings = sum(1 for r in all_results if r.severity == Severity.WARNING)

    if errors > 0 and config.validation.fail_on == "error":
        click.echo(f"  Validation failed: {errors} errors, {warnings} warnings")
        for r in all_results:
            if r.severity == Severity.ERROR:
                click.echo(f"    ERROR  {r.message}")
        sys.exit(1)

    # 4. Build manifest
    click.echo("  Building manifest...")
    manifest = build_manifest(config, data, sidecars, all_results)

    # 5. Render site
    click.echo("  Rendering site...")
    render_site(manifest, output_dir)

    # 6. Serialize ontology
    click.echo("  Serializing ontology...")
    serialize_ontology(ontology_path, output_dir, config.conneg.formats)

    # 7. Process diagrams
    figures_dir = config.resolve_path(config.figures)
    diagrams = collect_diagrams(figures_dir)
    if diagrams:
        click.echo(f"  Processing {len(diagrams)} diagrams...")
        diagrams_dir = output_dir / "diagrams"
        diagrams_dir.mkdir(exist_ok=True)
        term_names = {c.local_name for c in data.classes} | {p.local_name for p in data.properties}
        import shutil
        for diag in diagrams:
            if diag.suffix.lower() == ".svg":
                svg_content = diag.read_text()
                injected = inject_links(svg_content, term_names, base_path="../classes")
                (diagrams_dir / diag.name).write_text(injected)
            else:
                shutil.copy2(diag, diagrams_dir / diag.name)

    # 8. Content negotiation
    if config.conneg.generate:
        click.echo("  Generating content negotiation configs...")
        generate_conneg(output_dir, config.conneg.generate, config.conneg.base_uri, config.conneg.formats)

    click.echo(f"  Done! Output in {output_dir}/")
    if warnings > 0:
        click.echo(f"  ({warnings} warnings)")


@main.command()
@click.option("--config", "config_path", default="blathers.yaml", help="Path to blathers.yaml")
def fetch(config_path: str) -> None:
    """Pre-fetch and cache imported ontologies."""
    from blathers.config import load_config
    from blathers.imports import ImportResolver

    config = load_config(Path(config_path))
    if not config.imports:
        click.echo("No imports configured.")
        return

    cache_dir = config._config_dir / ".blathers" / "imports"
    resolver = ImportResolver(config.imports, config._config_dir, cache_dir)

    for imp in config.imports:
        click.echo(f"  Fetching {imp.uri}...", nl=False)
        success = resolver.fetch_and_cache(imp)
        click.echo(" OK" if success else " FAILED")
```
